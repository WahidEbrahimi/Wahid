
package exercise3_6;

import java.util.Scanner;
public class Exercise3_6 {

   
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        System.out.print("Enter the weight in kilogram: ");
        double wieght = input.nextDouble();
        System.out.print("Enter the height in meters: ");
        double hieght = input.nextDouble();
        double BMI = wieght / (hieght * hieght);
        System.out.printf("your bmi is:%.2f%n",BMI);
        if (BMI <= 18.5)
            System.out.println("inter pretation:under wieght");
        if (BMI <= 25)
            System.out.println("inter pretation:Normal");
        if (BMI <= 30)
            System.out.println("inter pretation:over wieght");
        else
            System.out.println("inter pretation:Obese");
            
    }
    
}
